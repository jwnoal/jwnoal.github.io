```
function getDevice() {
  const userAgent = navigator.userAgent.toLowerCase();
  if (
    userAgent.includes("iphone") ||
    userAgent.includes("ipad") ||
    userAgent.includes("ipod")
  ) {
    // 用户在 iOS 设备上
    console.log("iOS");
    return "ios";
  } else if (userAgent.includes("android")) {
    // 用户在 Android 设备上
    console.log("Android");
    return "Android";
  } else {
    // 用户在其他设备上，如桌面浏览器
    console.log("其他设备");
  }
}
```
